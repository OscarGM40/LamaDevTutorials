  const handleOption = (option: "adult" | "children" | "room",op:string) => {
    setOptions( prev => {
      return {
        ...prev,
        /* va a girar sobre la puta opción(me dió igual que fuera una enum) */
         [option]: op === 'add' ? prev[option] + 1 : prev[option] - 1
      }
    })
  }
